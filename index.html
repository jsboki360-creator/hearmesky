// ======================= // ✨ Canvas 설정 // ======================= const canvas = document.getElementById("c"); const ctx = canvas.getContext("2d"); canvas.width = window.innerWidth; canvas.height = window.innerHeight; // ======================= // 💬 메시지 받기 // ======================= const params = new URLSearchParams(location.search); const text = params.get("msg") || "Gina"; // ======================= // 🧠 텍스트 → 픽셀 // ======================= const off = document.createElement("canvas"); const offCtx = off.getContext("2d"); off.width = canvas.width; off.height = canvas.height; offCtx.fillStyle = "white"; offCtx.font = "bold 120px Arial"; const metrics = offCtx.measureText(text); const tx = (canvas.width - metrics.width) / 2; const ty = canvas.height / 2; offCtx.fillText(text, tx, ty); // ======================= // ✨ 입자 생성 // ======================= let particles = []; const imgData = offCtx.getImageData(0,0,off.width,off.height); for(let y=0;y 128){ particles.push({ x: Math.random()*canvas.width, y: Math.random()*canvas.height, tx: x, ty: y }); } } } // ======================= // 🎬 애니메이션 // ======================= function draw(){ ctx.clearRect(0,0,canvas.width,canvas.height); ctx.fillStyle="white"; particles.forEach(p=>{ p.x += (p.tx - p.x)*0.05; p.y += (p.ty - p.y)*0.05; ctx.beginPath(); ctx.arc(p.x,p.y,2,0,Math.PI*2); ctx.fill(); }); requestAnimationFrame(draw); } draw();
